#include <Wire.h>
#include <SPI.h>
#include <SD.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <DHT.h>

// Pin setup
#define DHTPIN 2
#define DHTTYPE DHT22

#define SD_CS_PIN 10

// OLED setup
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1

// Objects
DHT dht(DHTPIN, DHTTYPE);
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

// Logger setup
const char *fileName = "data.csv";
const unsigned long LOG_INTERVAL = 10000; // 10 seconds

unsigned long lastLogTime = 0;
bool firstReading = true;

void showMessage(String line1, String line2 = "", String line3 = "") {
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);

  display.setCursor(0, 0);
  display.println(line1);

  display.setCursor(0, 18);
  display.println(line2);

  display.setCursor(0, 36);
  display.println(line3);

  display.display();
}

void setup() {
  Serial.begin(9600);

  dht.begin();

  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("OLED failed");
    while (true);
  }

  showMessage("Data Logger", "Starting...");

  if (!SD.begin(SD_CS_PIN)) {
    Serial.println("SD card failed");
    showMessage("SD card failed", "Check wiring", "CS = D10");
    while (true);
  }

  Serial.println("SD card ready");

  if (!SD.exists(fileName)) {
    File dataFile = SD.open(fileName, FILE_WRITE);

    if (dataFile) {
      dataFile.println("time_seconds,temperature_c,humidity_percent");
      dataFile.close();
    }
  }

  showMessage("Data Logger", "SD card ready", "Logging...");
  delay(1000);
}

void loop() {
  unsigned long currentTime = millis();

  if (firstReading || currentTime - lastLogTime >= LOG_INTERVAL) {
    firstReading = false;
    lastLogTime = currentTime;

    float humidity = dht.readHumidity();
    float temperature = dht.readTemperature();

    if (isnan(humidity) || isnan(temperature)) {
      Serial.println("DHT22 read failed");
      showMessage("Sensor error", "DHT22 read failed");
      return;
    }

    unsigned long timeSeconds = currentTime / 1000;

    File dataFile = SD.open(fileName, FILE_WRITE);

    if (dataFile) {
      dataFile.print(timeSeconds);
      dataFile.print(",");
      dataFile.print(temperature, 1);
      dataFile.print(",");
      dataFile.println(humidity, 1);
      dataFile.close();

      Serial.print("Logged: ");
      Serial.print(timeSeconds);
      Serial.print("s, ");
      Serial.print(temperature, 1);
      Serial.print(" C, ");
      Serial.print(humidity, 1);
      Serial.println(" %");
    } else {
      Serial.println("Could not open data.csv");
      showMessage("SD write error", "Could not open", "data.csv");
      return;
    }

    display.clearDisplay();
    display.setTextColor(SSD1306_WHITE);

    display.setTextSize(1);
    display.setCursor(0, 0);
    display.println("DHT22 Data Logger");

    display.setCursor(0, 12);
    display.print("Time: ");
    display.print(timeSeconds);
    display.println(" sec");

    display.setTextSize(2);
    display.setCursor(0, 28);
    display.print(temperature, 1);
    display.println(" C");

    display.setCursor(0, 48);
    display.print(humidity, 1);
    display.println(" %");

    display.display();
  }
}
